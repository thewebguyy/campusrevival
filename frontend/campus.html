async function init() { const urlParams = new URLSearchParams(window.location.search); const schoolId = urlParams.get('id'); const slug = urlParams.get('slug'); if (!schoolId && !slug) { window.location.href = 'map.html'; return; } try { let school; if (schoolId) { school = await getSchoolById(schoolId); } else { // Add API for slug lookup later school = await getSchoolBySlug(slug); } if (!school) throw new Error('School not found'); // Update UI document.getElementById('school-name').textContent = school.name; document.getElementById('school-city').innerHTML = ` ${school.city || school.address}`; document.getElementById('adoption-count').textContent = school.adoptionCount || 0; document.getElementById('school-desc').textContent = school.description || "Building a legacy of faith and revival on this campus."; document.getElementById('page-title').textContent = `${school.name} | Campus Revival Movement`; // Update Meta Tags for SEO document.getElementById('meta-desc').content = `Join the movement praying for ${school.name}. Currently has ${school.adoptionCount} adopters.`; document.getElementById('og-title').content = `Pray for ${school.name}`; // Street View Logic const svBtn = document.getElementById('streetview-btn'); svBtn.onclick = () => { const url = `https://www.google.com/maps/@?api=1&map_action=pano&viewpoint=${school.lat},${school.lng}`; window.open(url, '_blank'); }; // Hero Background if (school.image) { document.getElementById('hero-bg').style.backgroundImage = `linear-gradient(to bottom, rgba(0,0,0,0.3), var(--bg-dark)), url('${school.image}')`; } // Load Partners if (school.partnerOrganizations && school.partnerOrganizations.length > 0) { const partnersList = document.getElementById('partners-list'); partnersList.innerHTML = school.partnerOrganizations.map(p => `
`).join(''); } // Load Prayer Requests loadPrayerRequests(school._id); loadImpactReport(school._id); } catch (err) { console.error(err); alert('Could not load institution details.'); } } async function loadPrayerRequests(schoolId) { try { const response = await fetch(`${API_URL}/prayer-requests/${schoolId}`); const data = await response.json(); const container = document.getElementById('board-requests'); const leadersContainer = document.getElementById('verified-leaders'); if (data.success && data.requests.length > 0) { container.innerHTML = data.requests.map(req => `
${req.isUrgent ? 'URGENT' : ''}

${req.content}

${req.userId.name} ${req.userId.isVerifiedLeader ? ' Verified' : ''} • ${new Date(req.createdAt).toLocaleDateString()}
`).join(''); // Populate leaders const uniqueLeaders = Array.from(new Set(data.requests.filter(r => r.userId.isVerifiedLeader).map(r => r.userId._id))) .map(id => data.requests.find(r => r.userId._id === id).userId); if (uniqueLeaders.length > 0) { leadersContainer.innerHTML = uniqueLeaders.map(l => `
${l.name.charAt(0)}
${l.name}
${l.organization || 'Campus Leader'}
`).join(''); } } } catch (err) { console.error('Error loading board:', err); } } async function loadImpactReport(schoolId) { try { const response = await fetch(`${API_URL}/schools/${schoolId}/impact`); const data = await response.json(); if (data.success && (data.report.newAdoptions > 0 || data.report.newJournals > 0)) { const card = document.getElementById('impact-card'); const stats = document.getElementById('impact-stats'); card.style.display = 'block'; stats.innerHTML = ` This month (${data.report.month}):
• ${data.report.newAdoptions} new advocates
• ${data.report.newJournals} testimonies recorded
${data.report.answeredPrayers > 0 ? `• ${data.report.answeredPrayers} answered prayers!` : ''} `; } } catch (err) { console.error('Impact report error:', err); } } function adopt(type) { const urlParams = new URLSearchParams(window.location.search); const schoolId = urlParams.get('id'); window.location.href = `adopt-${type}.html?schoolId=${schoolId}`; } window.onload = init;
